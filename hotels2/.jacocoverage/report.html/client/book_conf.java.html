    private void submitActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_submitActionPerformed
        // EISAGWGI STIN VASI
        stuff.main mn = new main();
        conn=mn.connect(conn);
        String cd = code.getText();
        int code=Integer.parseInt(cd);
        
        String lsnm=lastname.getText();
        String fsnm=firstname.getText();
        String reg=region.getText();
        String cat=category.getText();
        
        String chi=chin.getText();
        Date indate=Date.valueOf(chi);
        indate=convertUtilToSql(indate);
        
        String cho=chout.getText();
        Date otdate=Date.valueOf(cho);
        otdate=convertUtilToSql(otdate);
        
        int num=hotbooknum;
        String chrg=charge.getText();
        int charg=Integer.parseInt(chrg);
        
        try{
            String inbsql="insert into BOOKS values(?,?,?,?,?,?,?,?,?)";
            ptst=conn.prepareStatement(inbsql);
            ptst.setInt(1, code);
            ptst.setString(2, lsnm);
            ptst.setString(3, fsnm);
            ptst.setString(4, reg);
            ptst.setInt(5, hotbooknum);
            ptst.setDate(6, indate);
            ptst.setDate(7, otdate);
            ptst.setInt(8, charg);
            ptst.setString(9, cat);
            ptst.execute();
            JOptionPane.showMessageDialog(null, "Επιτυχης Κρατηση");
        }catch(Exception e){
            JOptionPane.showMessageDialog(null, e, "", JOptionPane.ERROR_MESSAGE);
        }
        
        
    }//GEN-LAST:event_submitActionPerformed
