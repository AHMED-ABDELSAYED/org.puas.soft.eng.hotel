    private void submitActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_submitActionPerformed
     // EISAGWGI STIN VASI PROSFORWN
        main mn = new main();
        conn=mn.connect(conn);
        String hotel=(String) area.getSelectedItem();
        String rtype=(String) type.getSelectedItem();
        String np=newprice.getText();
        if(np.isEmpty()){
            JOptionPane.showMessageDialog(null, "Παρακαλω συμπληρωστε τα πεδια", "", JOptionPane.ERROR_MESSAGE);
            return;
        }
        int newp=Integer.parseInt(np);
        Date fdate = fromcal.getDate();
        Date tdate = tocal.getDate();
        java.sql.Date fDate = convertUtilToSql(fdate);
        java.sql.Date tDate = convertUtilToSql(tdate);
        if(fdate.after(tdate)){
            JOptionPane.showMessageDialog(null, "Η τελικη ημερομηνια δεν μπορει να ειναι πριν την αρχικη", "", JOptionPane.ERROR_MESSAGE);
            return;
        }
        //EISAFWGI STIN VASI
        try{
            String isql="insert into OFFERS values(?,?,?,?,?)";
            pst=conn.prepareStatement(isql);
            pst.setString(1,hotel);
            pst.setString(2,rtype);
            pst.setDate(3,fDate);
            pst.setDate(4,tDate);
            pst.setInt(5,newp);
            pst.execute();
            JOptionPane.showMessageDialog(null, "Eπιτυχης εισαγωγη προσφορας");
        }catch(Exception e){
            JOptionPane.showMessageDialog(null, e, "", JOptionPane.ERROR_MESSAGE);
        }
    }//GEN-LAST:event_submitActionPerformed
