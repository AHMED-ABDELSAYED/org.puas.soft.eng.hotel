    private void submitActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_submitActionPerformed
        //ALLAGI KWDIKOU PROSVASIS
        main mn = new main();
        conn=mn.connect(conn);
        String usn=usnm.getText();
        String nps=npasswd.getText();
        String cps=ncpasswd.getText();
        if(usn.isEmpty() || nps.isEmpty() || cps.isEmpty()){
            JOptionPane.showMessageDialog(null,"Παρακαλω συμπληρωστε τηην φορμα","",JOptionPane.ERROR_MESSAGE);
            return;
        }else{
            if(!nps.equals(cps)){
               JOptionPane.showMessageDialog(null, "Οι κωδικοι δεν ταιριαζουν", "", JOptionPane.ERROR_MESSAGE);
               return;
            }
            try{
                String chsql="update ACCOUNT set PASSWD=? where USERNAME=?";
                pst=conn.prepareStatement(chsql);
                pst.setString(1,nps);
                pst.setString(2,usn);
                pst.execute();
                JOptionPane.showMessageDialog(null, "Επιτυχης αλλαγη κωδικου");
            }catch(Exception e){
                JOptionPane.showMessageDialog(null, e, "", JOptionPane.ERROR_MESSAGE);
            }
        }
    }//GEN-LAST:event_submitActionPerformed
