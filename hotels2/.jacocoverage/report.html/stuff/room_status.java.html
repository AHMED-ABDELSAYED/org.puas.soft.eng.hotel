    public ArrayList<Avail> availList(){
        conn=main.connect(conn);
        ArrayList<Avail> availList = new ArrayList<>();
        try{
            String ssql="select * from HOTEL";
            pst=conn.prepareStatement(ssql);
            rs=pst.executeQuery();
            Avail avail;
            while(rs.next()){
                avail=new Avail(rs.getInt("ROOM_NO"),rs.getString("CATEGORY"),rs.getString("REGION"),rs.getInt("PRICE"),rs.getBoolean("AVAILABLE"));
                availList.add(avail);
            }
        }catch(Exception e){
            e.printStackTrace();
        }
        return availList;
    }
    protected void clear_table(){
        rstate.setModel(new DefaultTableModel(null,new String[]{"Κωδικος","Τυπος","Περιοχη","Τιμη","Διαθεσιμο"}));  
    }
    
    public void showAvail(){
        ArrayList<Avail> avail = availList();
         DefaultTableModel model = (DefaultTableModel) rstate.getModel();
          Object[] row = new Object[5];
        for(int i=0;i<avail.size();i++){
            row[0]=avail.get(i).getRoom_no();
            row[1]=avail.get(i).getType();
            row[2]=avail.get(i).getRegion();
            row[3]=avail.get(i).getPrice();
            row[4]=avail.get(i).isAvail();          
            model.addRow(row);
        }
    }

    private void chstateActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_chstateActionPerformed
        // ENIMERWSI VASIS KAI UPDATE TO JTABLE
        int uproom=rstate.getSelectedRow();
        if(uproom==-1){
            JOptionPane.showMessageDialog(null, "Παρακαλω διαλεξτε μια εγγραφη", "", JOptionPane.ERROR_MESSAGE);
        }else{
            int upcode=(int) rstate.getModel().getValueAt(uproom, 0);
            boolean upstate=(boolean) rstate.getModel().getValueAt(uproom,4);
            try{
                if(upstate==true){
                    String upsql="update HOTEL set AVAILABLE=false where ROOM_NO=?";
                    pst=conn.prepareStatement(upsql);
                    pst.setInt(1, upcode);
                    pst.execute();
                }else{
                    String upsql="update HOTEL set AVAILABLE=true where ROOM_NO=?";
                    pst=conn.prepareStatement(upsql);
                    pst.setInt(1, upcode);
                    pst.execute();
                }
            }catch(Exception e){
                e.printStackTrace();
            }
        }
        clear_table();
        showAvail();
    }//GEN-LAST:event_chstateActionPerformed

    private void deleteroomActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_deleteroomActionPerformed
        //DIAGRAGI DWMATIOU
        int delrow=rstate.getSelectedRow();
        if(delrow==-1){
            JOptionPane.showMessageDialog(null,"Παρακαλω διαλεξτε μια στηλη","",JOptionPane.ERROR_MESSAGE);
            return;
        }else{
            int delchoise=JOptionPane.showConfirmDialog(null, "Ειστε σιγουροι ?", "", JOptionPane.YES_NO_OPTION);
            if(delchoise==0){
                  int delroomno=(int)rstate.getValueAt(delrow, 0);
            String delreg=(String)rstate.getValueAt(delrow,2);
            try{
                String delsql="delete from HOTEL where ROOM_NO=? and REGION=?";
                pst=conn.prepareStatement(delsql);
                pst.setInt(1,delroomno);                
                pst.setString(2,delreg);

                clear_table();
