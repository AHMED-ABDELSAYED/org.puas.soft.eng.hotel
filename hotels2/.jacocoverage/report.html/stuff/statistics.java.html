    private void createGraphsActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_createGraphsActionPerformed
        //ELEGXOS IMEROMINIAS KAI PARAGWGI GRAFIMATWN
        DefaultCategoryDataset dataset = new DefaultCategoryDataset();
        
        main mn = new main();
        conn=mn.connect(conn);
        int ch=choise.getSelectedIndex();
        Date stdt = convertUtilToSql(stdate.getDate());
        Date endt = convertUtilToSql(endate.getDate());
        if(endt.before(stdt)){
            JOptionPane.showMessageDialog(null, "Η τελικη ημερομηνια δεν μπορει να ειναι πριν την αρχικη", "", JOptionPane.ERROR_MESSAGE);
            return;
        }
        java.sql.Date rng = stdt;
        if(ch==0){
            while(rng.before(endt)){
                int cnt=0;
                try{
                    String selsql="select distinct * from  BOOKS where ? between CHECKIN and CHECKOUT";
                    ptst=conn.prepareStatement(selsql);
                    ptst.setDate(1, rng);
                    rs=ptst.executeQuery();
                    while(rs.next()){
                       cnt++; 
                    }
                    dataset.setValue(cnt,""," "+rng); 
                
                    //System.out.println(cnt);
                    rng.setDate(rng.getDate()+1);                              
                }catch(Exception e){
                    e.printStackTrace();
                }                 
            }
            JFreeChart chart = ChartFactory.createBarChart("ΓΡΑΦΙΚΗ","ΗΜΕΡΟΜΗΝΙΕΣ","ΚΡΑΤΗΣΕΙΣ",dataset,PlotOrientation.VERTICAL, false, true,false);
            CategoryPlot p = chart.getCategoryPlot();
            ChartFrame frame = new ChartFrame("TEST CHART",chart);
            frame.setVisible(true);
            frame.setSize(600, 600);
            frame.setLocationRelativeTo(null); 
        }else{
            String hchoise=(String)choise.getSelectedItem();
            while(rng.before(endt)){
                int cnt=0;
                try{
                    String selsql="select distinct * from  BOOKS where REGION=? and ? between CHECKIN and CHECKOUT";
                    ptst=conn.prepareStatement(selsql);
                    ptst.setString(1, hchoise);
                    ptst.setDate(2, rng);
                    rs=ptst.executeQuery();
                    while(rs.next()){
                       cnt++; 
                    }
                    dataset.setValue(cnt,""," "+rng); 
                
                    System.out.println(cnt);
                    rng.setDate(rng.getDate()+1);                              
                }catch(Exception e){
                    e.printStackTrace();
                }
            }
            JFreeChart chart = ChartFactory.createBarChart("ΓΡΑΦΙΚΗ","ΗΜΕΡΟΜΗΝΙΕΣ","ΚΡΑΤΗΣΕΙΣ",dataset,PlotOrientation.VERTICAL, false, true,false);
            CategoryPlot p = chart.getCategoryPlot();
            ChartFrame frame = new ChartFrame("TEST CHART",chart);
            frame.setVisible(true);
            frame.setSize(600, 600);
            frame.setLocationRelativeTo(null); 
        }   
    }//GEN-LAST:event_createGraphsActionPerformed
