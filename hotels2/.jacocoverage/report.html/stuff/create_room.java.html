    private void submitActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_submitActionPerformed
        //EISAGWGI STIN VASI TO DWMATIO
        main mn = new main();
        conn=mn.connect(conn);
        
        String hot=(String) hotel.getSelectedItem();
        String typ=(String) type.getSelectedItem();
        
        String roomn = roomno.getText();
        if(roomn.isEmpty()){
            JOptionPane.showMessageDialog(null, "Παρακαλω συμπληρωστε τα πεδια", "", JOptionPane.ERROR_MESSAGE);
            return;
        }
        int roomnum=Integer.parseInt(roomn);
        
        String prc=price.getText();
        if(prc.isEmpty()){
            JOptionPane.showMessageDialog(null, "Παρακαλω συμπληρωστε τα πεδια", "", JOptionPane.ERROR_MESSAGE);
            return;
        }
        int pric=Integer.parseInt(prc);
        
        try{
            String inssql="insert into HOTEL values(?,?,?,?,?)";
            ptst=conn.prepareStatement(inssql);
            ptst.setInt(1,roomnum);
            ptst.setString(2,hot);
            ptst.setString(3,typ);
            ptst.setInt(4,pric);
            ptst.setBoolean(5,true);
            ptst.execute();
            JOptionPane.showMessageDialog(null, "Επιτυχης Καταχωρηση");
        }catch(Exception e){
            JOptionPane.showMessageDialog(null,e,"",JOptionPane.ERROR_MESSAGE);
        }
    }//GEN-LAST:event_submitActionPerformed
